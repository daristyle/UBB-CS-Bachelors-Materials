 12def create_expense(apartment: int, amount: int, type:str) -> list:
 13    """
 14    :param apartment: number of the apartment
 15    :param amount: the amount
 16    :param type: the type of the expenses
 17    :return: list, representing the expense
 18    Raises an error if:
 19        - the apartment number is not a positive integer
 20        - the amount is negative
 21        - the type of the expenses is not water, heating, electricity, gas or other
 22    """
 23    if apartment <= 0:
 24        raise ValueError("Apartment number must be positive")
 25    if amount < 0:
 26        raise ValueError("Amount must be positive")
 27    if type not in ["water", "heating", "electricity", "gas", "other"]:
 28        raise ValueError("Type must be one of water, heating, gas or other")
 29
 30    return [apartment, amount, type]

105def remove_apartments(l: list, first: int, last: int) -> None:
106    """
107    :param l: list of expenses
108    :param first: the number of the first apartment that needs to be removed
109    :param last: the number of the last apartment that needs to be removed

120    l_new = []
121    for x in l:

123            l_new.append(x)
124
125    l[:] = l_new
