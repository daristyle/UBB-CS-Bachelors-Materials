EN: Give values for T, N1, N2, N3 (all strictly positive) for which the program below finishes correctly. In each loop iteration, it creates: 1 thread for f1 (waits on b1), 2 threads for f2 (waits on b2), and 3 threads for f3 (waits on b3).
 
#include <pthread.h>
 
pthread_barrier_t b1, b2, b3;
 
void* f1(void* a) { pthread_barrier_wait(&b1); return NULL; } void* f2(void* a) { pthread_barrier_wait(&b2); return NULL; } void* f3(void* a) { pthread_barrier_wait(&b3); return NULL; }
 
int main() {     int T, N1, N2, N3;
 
    pthread_t all_threads[6 * T];
 
    pthread_barrier_init(&b1, NULL, N1);     pthread_barrier_init(&b2, NULL, N2);     pthread_barrier_init(&b3, NULL, N3);
 
    for(int i=0; i<T; i++) {         int base_idx = i * 6;         pthread_create(&all_threads[base_idx + 0], NULL, f1, NULL);         pthread_create(&all_threads[base_idx + 1], NULL, f2, NULL);         pthread_create(&all_threads[base_idx + 2], NULL, f2, NULL);         pthread_create(&all_threads[base_idx + 3], NULL, f3, NULL);         pthread_create(&all_threads[base_idx + 4], NULL, f3, NULL);         pthread_create(&all_threads[base_idx + 5], NULL, f3, NULL);     }
 
    for(int i=0; i<6*T; i++) {         pthread_join(all_threads[i], NULL);     }
 
    pthread_barrier_destroy(&b1);     pthread_barrier_destroy(&b2);     pthread_barrier_destroy(&b3);     return 0; }
